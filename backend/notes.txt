# 1. Create the backend folder
mkdir backend
cd backend

# 2. Create a virtual environment to keep dependencies isolated
python -m venv venv

# 3. Activate the environment
# For Windows:
.\venv\Scripts\activate
# For Mac/Linux:
source venv/bin/activate

pip install django djangorestframework django-cors-headers django-environ celery redis

# 1. Start the project (we call the config folder 'core')
django-admin startproject core .

# 2. Create the apps directory and make it a Python package
mkdir apps
# Windows: type nul > apps\__init__.py | Mac/Linux: touch apps/__init__.py
touch apps/__init__.py

# 3. Create our specific feature apps inside the apps folder
python manage.py startapp accounts apps/accounts
python manage.py startapp courses apps/courses
python manage.py startapp tasks apps/tasks
python manage.py startapp resources apps/resources
python manage.py startapp ai_lab apps/ai_lab
